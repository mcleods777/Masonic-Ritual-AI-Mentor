let inactivityListenerAttached = false;

2. **Add the timer-reset function (module-scope):**
```typescript
function resetInactivityTimer(): void {
  if (inactivityTimer) clearTimeout(inactivityTimer);
  if (!desired) return; // lock not active — no timer needed
  inactivityTimer = setTimeout(() => {
    void releaseSentinel();
    desired = false; // prevents visibilitychange re-acquire per CONTEXT "do NOT auto-reacquire"
    console.info("[SAFETY-07] Wake lock released after 30 min of inactivity");
  }, INACTIVITY_TIMEOUT_MS);
}
```

(If `releaseSentinel` is named differently in the current file, adapt. Read first.)

3. **Add the listener-attach function (idempotent, same pattern as `attachVisibilityListener`):**
```typescript
function attachInactivityListener(): void {
  if (inactivityListenerAttached) return;
  if (typeof document === "undefined") return;
  for (const ev of INACTIVITY_EVENTS) {
    document.addEventListener(ev, resetInactivityTimer, { passive: true });
  }
  inactivityListenerAttached = true;
}
```

```tsx
// @vitest-environment jsdom
import { afterEach, beforeEach, describe, expect, it, vi } from "vitest";

describe("screen-wake-lock SAFETY-07 inactivity release", () => {
  let releaseSpy: ReturnType<typeof vi.fn>;

  beforeEach(() => {
    vi.resetModules();
    vi.useFakeTimers();
    releaseSpy = vi.fn().mockResolvedValue(undefined);
    Object.defineProperty(navigator, "wakeLock", {
      configurable: true,
      value: { request: vi.fn().mockResolvedValue({ release: releaseSpy, addEventListener: vi.fn() }) },
    });
  });
  afterEach(() => {
    vi.useRealTimers();
    vi.restoreAllMocks();
  });

  it("acquires wake lock on keepScreenAwake", async () => {
    const { keepScreenAwake } = await import("../screen-wake-lock");
    await keepScreenAwake();
    expect((navigator as any).wakeLock.request).toHaveBeenCalledWith("screen");
  });

  it("releases after 30 min of inactivity", async () => {
    const infoSpy = vi.spyOn(console, "info").mockImplementation(() => {});
    const { keepScreenAwake } = await import("../screen-wake-lock");
    await keepScreenAwake();
    vi.advanceTimersByTime(30 * 60 * 1000);
    await vi.runAllTicks();
    expect(releaseSpy).toHaveBeenCalled();
    expect(infoSpy).toHaveBeenCalledWith(expect.stringContaining("[SAFETY-07]"));
  });

  it("keydown before 30min resets timer", async () => {
    const { keepScreenAwake } = await import("../screen-wake-lock");
    await keepScreenAwake();
    vi.advanceTimersByTime(29 * 60 * 1000);
    document.dispatchEvent(new Event("keydown"));
    vi.advanceTimersByTime(29 * 60 * 1000);
    // Still < 30min since last keydown → should not release yet
    expect(releaseSpy).not.toHaveBeenCalled();
  });

  // ... Tests 4-6 covering post-release behavior + idempotent attach
});
```
