<interfaces>
CONTEXT D-14 locked behavior:
- Middleware verifies shared-secret + client-token on `/api/*` (except `/api/auth/*`)
- Each paid-route handler ALSO calls `requireClientToken` (via applyPaidRouteGuards) before rate-limit
- Either check failing → 401 + `{error:'client_token_invalid'}`

The route-level check is implemented inside `applyPaidRouteGuards` (Plan 05 extension). When a paid route handler runs `const guard = await applyPaidRouteGuards(request, {routeName})`, the guard re-verifies the Bearer via `verifyClientToken` and returns a `{kind:"deny", response}` with 401 if invalid.

SAFETY-09 test strategy (the "middleware bypass" simulation):
- Invoke the route handler DIRECTLY via `const resp = await POST(request)` — this skips Next.js middleware entirely (unit tests do this naturally; middleware is invoked by the Next.js router, not by calling route handlers directly)
- Construct a NextRequest WITHOUT an Authorization header (and without a valid cookie)
- Assert the response is 401 + `{error:"client_token_invalid"}`

This proves that even if middleware never ran (e.g., a future matcher regression excluded the route, or a server action skipped middleware), the route handler's own call to applyPaidRouteGuards is the last line of defense.
</interfaces>

```typescript
// @vitest-environment node
/**
 * SAFETY-09 defense-in-depth regression test.
 *
 * Proves that direct invocation of POST (simulating a future Next.js
 * middleware-skip quirk) is still blocked by the route's own
 * applyPaidRouteGuards → verifyClientToken call. Middleware is the
 * perimeter; this test proves the route is its own perimeter too.
 *
 * If this test ever fails: a paid route can be reached without a valid
 * Bearer. Immediate incident — flip the kill switch per
 * docs/runbooks/KILL-SWITCH.md while investigating.
 */

import { afterEach, beforeEach, describe, expect, it, vi } from "vitest";
import { NextRequest } from "next/server";
import { __resetRateLimitForTests } from "@/lib/rate-limit";
import { signClientToken, signSessionToken } from "@/lib/auth";
import { POST } from "../route";

beforeEach(() => {
  __resetRateLimitForTests();
  delete process.env.RITUAL_EMERGENCY_DISABLE_PAID;
  process.env.JWT_SECRET = "x".repeat(32);
});
afterEach(() => {
  vi.restoreAllMocks();
});

function makeRequest(headers: Record<string, string>, body: Record<string, unknown>): NextRequest {
  return new NextRequest("https://test.local/api/tts/gemini", {
    method: "POST",
    headers: { "content-type": "application/json", ...headers },
    body: JSON.stringify(body),
  });
}

describe("SAFETY-09 tts/gemini defense-in-depth", () => {
  it("rejects direct invocation with NO Authorization header (401 + client_token_invalid)", async () => {
    const req = makeRequest({}, { text: "test" });
    const resp = await POST(req);
    expect(resp.status).toBe(401);
    const body = await resp.json();
    expect(body).toEqual({ error: "client_token_invalid" });
  });

  it("rejects invalid Bearer token", async () => {
    const req = makeRequest({ authorization: "Bearer not-a-valid-jwt" }, { text: "test" });
    const resp = await POST(req);
    expect(resp.status).toBe(401);
  });

  it("rejects cross-audience token (session token presented as Bearer)", async () => {
    const sessionToken = await signSessionToken("brother@example.com");
    const req = makeRequest({ authorization: `Bearer ${sessionToken}` }, { text: "test" });
    const resp = await POST(req);
    expect(resp.status).toBe(401);
  });

  it("passes the guard with a valid client-token Bearer (no client_token_invalid)", async () => {
    // Mock the upstream fetch so the test doesn't actually hit Gemini.
    vi.spyOn(globalThis, "fetch").mockImplementation(async (url: string | URL | Request) => {
      const urlStr = typeof url === "string" ? url : url instanceof URL ? url.toString() : url.url;
      if (urlStr.includes("generativelanguage.googleapis.com") || urlStr.includes("gemini")) {
        // Minimal Gemini-shaped response to let the route proceed past the upstream call
        return new Response(JSON.stringify({ candidates: [] }), { status: 200 });
      }
      return new Response("not found", { status: 404 });
    });
    const clientToken = await signClientToken("abc0123456789def");
    const req = makeRequest({ authorization: `Bearer ${clientToken}` }, { text: "test" });
    const resp = await POST(req);
    // We don't care if the route then returns 200/500/502 — we only care that it
    // did NOT return 401 client_token_invalid.
    if (resp.status === 401) {
      const body = await resp.json();
      expect(body.error).not.toBe("client_token_invalid");
    }
    expect(resp.status).not.toBe(401); // stricter: guard should have passed
  });
});
```
