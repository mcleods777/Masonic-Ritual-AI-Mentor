/**
 * Client-token (SAFETY-05, D-11/D-12). Short-lived 1h JWT issued to a
 * signed-in browser by POST /api/auth/client-token and ridden along with
 * every paid-route call as Authorization: Bearer <token>.
 *
 * Audience `client-token` is DISTINCT from the 30-day `pilot-session` cookie's
 * audience. A stolen session cookie cannot be replayed as a client-token
 * (jwtVerify's audience option rejects the mismatch), and vice versa.
 *
 * No stateful session ID inside claims — per D-11, Phase 2 keeps claims
 * stateless. ADMIN-04 (Phase 6) introduces stateful revocation alongside
 * the durable KV store.
 */
export async function signClientToken(hashedUser: string): Promise<string> {
  return new SignJWT({ sub: hashedUser })
    .setProtectedHeader({ alg: "HS256" })
    .setIssuedAt()
    .setIssuer(ISSUER)
    .setAudience(CLIENT_TOKEN_AUDIENCE)
    .setExpirationTime(`${CLIENT_TOKEN_TTL_SECONDS}s`)
    .sign(getSecret());
}

export async function verifyClientToken(
  token: string | undefined,
): Promise<{ sub: string } | null> {
  if (!token) return null;
  try {
    const { payload } = await jwtVerify(token, getSecret(), {
      issuer: ISSUER,
      audience: CLIENT_TOKEN_AUDIENCE,
    });
    const sub = payload.sub;
    if (typeof sub !== "string") return null;
    return { sub };
  } catch {
    return null;
  }
}
```

/**
 * Client-token issuance endpoint (SAFETY-05, D-11/D-12).
 *
 * POST only. Gates:
 *   1. Origin header (when present) matches ALLOWED_ORIGIN_SUFFIXES — otherwise 403.
 *      Absent Origin (same-origin non-fetch paths) is allowed.
 *   2. pilot-session cookie valid — otherwise 401.
 *
 * Returns 1h JWT with sub: hashedUserFromEmail(session.email). No
 * request-body parsing — all state comes from cookie + header.
 *
 * Middleware carve-out: this endpoint sits under /api/auth/ which is
 * the existing skip-shared-secret + skip-client-token path (see
 * src/middleware.ts isPilotPublicPath). Renaming breaks bootstrap —
 * see RESEARCH §Pitfall 1.
 */

import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";
import {
  SESSION_COOKIE_NAME,
  CLIENT_TOKEN_TTL_SECONDS,
  signClientToken,
  verifySessionToken,
} from "@/lib/auth";
import { hashedUserFromEmail } from "@/lib/hash-user";

export const runtime = "nodejs";

// Duplicated from src/middleware.ts lines 13-31 per PATTERNS §5.
// If Phase 3+ extracts src/lib/origin.ts, consolidate there.
const ALLOWED_ORIGIN_SUFFIXES = [
  "masonic-ritual-ai-mentor.vercel.app",
  "localhost:3000",
  "localhost:3001",
  "127.0.0.1:3000",
];

function isAllowedOrigin(origin: string | null): boolean {
  if (!origin) return true; // same-origin browsers may omit Origin
  try {
    const { host } = new URL(origin);
    return (
      ALLOWED_ORIGIN_SUFFIXES.some((s) => host === s) ||
      host.endsWith(".vercel.app")
    );
  } catch {
    return false;
  }
}

export async function POST(req: NextRequest) {
  // 1. Origin check (D-12)
  const origin = req.headers.get("origin");
  if (!isAllowedOrigin(origin)) {
    return NextResponse.json({ error: "Forbidden origin" }, { status: 403 });
  }

  // 2. Pilot session cookie gate (D-12)
  const cookie = req.cookies.get(SESSION_COOKIE_NAME)?.value;
  const session = await verifySessionToken(cookie);
  if (!session) {
    return NextResponse.json({ error: "Not signed in" }, { status: 401 });
  }

  // 3. Derive hashedUser + sign
  const hashedUser = hashedUserFromEmail(session.email);
  const token = await signClientToken(hashedUser);

  return NextResponse.json({ token, expiresIn: CLIENT_TOKEN_TTL_SECONDS });
}
```

```typescript
/**
 * Wrapper around `fetch` for calling this app's own /api routes.
 *
 * Attaches TWO independent headers (SAFETY-05, D-15):
 *
 *   - X-Client-Secret: build-time env var. Low-strength deterrent against
 *     drive-by curl abuse. Rotated via the SECRET-ROTATION runbook.
 *
 *   - Authorization: Bearer <1h-client-token>. Obtained from
 *     POST /api/auth/client-token on first fetchApi call. Refreshed
 *     proactively at 50min via setTimeout; reactively on a 401
 *     `client_token_expired` retry (one shot, no infinite loop).
 *
 * These are independent concerns: X-Client-Secret is a perimeter
 * speed-bump; Authorization is a per-session bearer tied to the
 * pilot-session cookie. Both headers ride every paid-route call.
 *
 * Background-tab throttling caveat (RESEARCH §Pitfall 5): browsers
 * throttle setTimeout in background tabs so the 50min proactive
 * refresh can fire anywhere from 50-90+ min later. The visibilitychange
 * listener resets the timer on tab resume; the reactive 401 retry is
 * the safety net.
 */

const CLIENT_SECRET = process.env.NEXT_PUBLIC_RITUAL_CLIENT_SECRET;
const REFRESH_MS = 50 * 60 * 1000; // 10min safety before 60min expiry

let clientToken: string | null = null;
let refreshTimer: ReturnType<typeof setTimeout> | null = null;
let bootstrapInFlight: Promise<string | null> | null = null;

async function fetchClientToken(): Promise<string | null> {
  try {
    const resp = await fetch("/api/auth/client-token", {
      method: "POST",
      credentials: "include",
    });
    if (!resp.ok) return null;
    const body = (await resp.json()) as { token?: string; expiresIn?: number };
    return typeof body.token === "string" ? body.token : null;
  } catch {
    return null;
  }
}

function scheduleRefresh(): void {
  if (refreshTimer) clearTimeout(refreshTimer);
  refreshTimer = setTimeout(async () => {
    const fresh = await fetchClientToken();
    if (fresh) {
      clientToken = fresh;
      scheduleRefresh();
    } else {
      clientToken = null; // Failed refresh — reactive retry path will handle
    }
  }, REFRESH_MS);
}

async function ensureToken(): Promise<string | null> {
  if (clientToken) return clientToken;
  if (!bootstrapInFlight) {
    bootstrapInFlight = fetchClientToken().then((tok) => {
      clientToken = tok;
      if (tok) scheduleRefresh();
      bootstrapInFlight = null;
      return tok;
    });
  }
  return bootstrapInFlight;
}

// visibilitychange listener (D-13 + RESEARCH Pitfall 5) — reset the
// timer on resume since browsers throttle setTimeout in background tabs.
if (typeof document !== "undefined") {
  document.addEventListener("visibilitychange", () => {
    if (document.visibilityState === "visible" && clientToken) {
      scheduleRefresh();
    }
  });
}

function withBothHeaders(init: RequestInit | undefined, token: string | null): RequestInit {
  const merged: RequestInit = { ...init };
  const headers = new Headers(init?.headers);
  if (CLIENT_SECRET) headers.set("X-Client-Secret", CLIENT_SECRET);
  if (token) headers.set("Authorization", `Bearer ${token}`);
  merged.headers = headers;
  return merged;
}

export async function fetchApi(
  input: RequestInfo | URL,
  init?: RequestInit,
): Promise<Response> {
  const token = await ensureToken();
  const enriched = withBothHeaders(init, token);
  const resp = await fetch(input, enriched);

  // Reactive 401 retry on expired client-token (one shot, no infinite loop)
  if (resp.status === 401) {
    const cloned = resp.clone();
    let errKind: string | undefined;
    try {
      const body = (await cloned.json()) as { error?: string };
      errKind = body.error;
    } catch {
      // not JSON — fall through
    }
    if (errKind === "client_token_expired" || errKind === "client_token_invalid") {
      clientToken = null;
      const fresh = await ensureToken();
      if (fresh) {
        return fetch(input, withBothHeaders(init, fresh));
      }
    }
  }
  return resp;
}

/** Test-only: clear in-memory state between test cases. */
export function __resetApiFetchForTests(): void {
  clientToken = null;
  if (refreshTimer) clearTimeout(refreshTimer);
  refreshTimer = null;
  bootstrapInFlight = null;
}
```
